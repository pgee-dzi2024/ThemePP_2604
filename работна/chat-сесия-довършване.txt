## Какво най-често подобрява резултата

### 1) По-силна предобработка на изображението
Само grayscale + autocontrast често не стига. Обикновено помагат:

- **увеличаване на изображението**
- **денойз**
- **изостряне**
- **бинаризация / threshold**
- **премахване на наклон**
- **правилна ориентация**
- **засилване на контраста**

Примерна посока:
```python
from PIL import Image, ImageOps, ImageFilter
import cv2
import numpy as np

def preprocess_for_ocr(img):
    img = img.convert("RGB")
    img = np.array(img)

    gray = cv2.cvtColor(img, cv2.COLOR_RGB2GRAY)

    # увеличение
    gray = cv2.resize(gray, None, fx=2, fy=2, interpolation=cv2.INTER_CUBIC)

    # denoise
    gray = cv2.fastNlMeansDenoising(gray, None, 30, 7, 21)

    # threshold
    gray = cv2.threshold(gray, 0, 255, cv2.THRESH_BINARY + cv2.THRESH_OTSU)[1]

    return Image.fromarray(gray)
```

---

### 2) Промяна на `psm`
`--psm 6` е добре за блок текст, но не е универсален.

Пробвай:
- `--psm 4` — ако има текст в колони / по-структуриран документ
- `--psm 6` — ако има един блок текст
- `--psm 11` — за „разпръснат“ текст
- `--psm 3` — автоматичен режим

Често за документи:
```python
config = '--oem 1 --psm 4'
```

---

### 3) Ограничаване на езиците
Ако документът не е реално двуезичен, по-добре **не използвай винаги `bul+eng`**.  
Понякога Tesseract „се колебае“ между езиците и грешките се увеличават.

- само български: `lang='bul'`
- само английски: `lang='eng'`
- ако е смесен документ: `lang='bul+eng'`

---

### 4) Увеличи резолюцията / качеството на входа
Ако снимката е:
- размазана
- правена под ъгъл
- с отблясъци
- с ниска резолюция

OCR резултатът ще е слаб, независимо от кода.

Добри практики:
- снимка поне **1500–2000 px** по дългата страна
- документът да е добре осветен
- да е изправен и да не е отдалечен
- да няма сенки по текста

---

### 5) Премахване на наклон и перспектива
При снимки от телефон това е много важно.  
Ако документът е леко завъртян или „изкривен“, Tesseract греши повече.

Можеш да добавиш:
- deskew
- perspective transform
- crop на самия документ преди OCR

---

### 6) Пост-обработка на текста
При български OCR често има грешки от типа:
- `е` ↔ `c`
- `о` ↔ `0`
- `л` ↔ `п`
- `ш` ↔ `щ`

Можеш да:
- почистваш излишни нови редове
- нормализираш интервали
- правиш речникова корекция за често срещани думи

```python
from django.shortcuts import render
from rest_framework.views import APIView
from rest_framework.response import Response
from rest_framework import status
from .serializers import DocumentSerializer
import pytesseract
from PIL import Image, ImageOps, ImageFilter, ImageEnhance
import cv2
import numpy as np

pytesseract.pytesseract.tesseract_cmd = r'C:\Program Files\Tesseract-OCR\tesseract.exe'


class DocumentUploadView(APIView):
    def _preprocess_image(self, img):
        img = img.convert("RGB")
        img = np.array(img)

        gray = cv2.cvtColor(img, cv2.COLOR_RGB2GRAY)

        # Увеличаване за по-добро OCR разпознаване
        gray = cv2.resize(gray, None, fx=2, fy=2, interpolation=cv2.INTER_CUBIC)

        # Премахване на шум
        gray = cv2.fastNlMeansDenoising(gray, None, 30, 7, 21)

        # Изостряне чрез контрастен threshold
        gray = cv2.threshold(gray, 0, 255, cv2.THRESH_BINARY + cv2.THRESH_OTSU)[1]

        return Image.fromarray(gray)

    def post(self, request, *args, **kwargs):

        if serializer.is_valid():

            try:
                img = Image.open(document.image.path)
                img = self._preprocess_image(img)

                config = '--oem 1 --psm 4'
                extracted_text = pytesseract.image_to_string(

                    config=config

                document.extracted_text = extracted_text
                document.save()

                return Response({

                return Response(
                    {'error': f'Грешка при обработката на изображението: {str(e)}'},
                    status=status.HTTP_500_INTERNAL_SERVER_ERROR
                )

        return Response(serializer.errors, status=status.HTTP_400_BAD_REQUEST)
```

```python
from django.shortcuts import render
from rest_framework.views import APIView
from rest_framework.response import Response
from rest_framework import status
from .serializers import DocumentSerializer
import pytesseract
from PIL import Image
import cv2
import numpy as np

pytesseract.pytesseract.tesseract_cmd = r'C:\Program Files\Tesseract-OCR\tesseract.exe'


class DocumentUploadView(APIView):
    def _order_points(self, pts):
        rect = np.zeros((4, 2), dtype="float32")
        s = pts.sum(axis=1)
        rect[0] = pts[np.argmin(s)]
        rect[2] = pts[np.argmax(s)]

        diff = np.diff(pts, axis=1)
        rect[1] = pts[np.argmin(diff)]
        rect[3] = pts[np.argmax(diff)]
        return rect

    def _four_point_transform(self, image, pts):
        rect = self._order_points(pts)
        (tl, tr, br, bl) = rect

        width_a = np.linalg.norm(br - bl)
        width_b = np.linalg.norm(tr - tl)
        max_width = max(int(width_a), int(width_b))

        height_a = np.linalg.norm(tr - br)
        height_b = np.linalg.norm(tl - bl)
        max_height = max(int(height_a), int(height_b))

        dst = np.array([
            [0, 0],
            [max_width - 1, 0],
            [max_width - 1, max_height - 1],
            [0, max_height - 1]
        ], dtype="float32")

        matrix = cv2.getPerspectiveTransform(rect, dst)
        warped = cv2.warpPerspective(image, matrix, (max_width, max_height))
        return warped

    def _correct_perspective(self, image):
        orig = image.copy()
        gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
        blurred = cv2.GaussianBlur(gray, (5, 5), 0)
        edged = cv2.Canny(blurred, 75, 200)

        contours, _ = cv2.findContours(edged.copy(), cv2.RETR_LIST, cv2.CHAIN_APPROX_SIMPLE)
        contours = sorted(contours, key=cv2.contourArea, reverse=True)[:5]

        for c in contours:
            peri = cv2.arcLength(c, True)
            approx = cv2.approxPolyDP(c, 0.02 * peri, True)

            if len(approx) == 4:
                return self._four_point_transform(orig, approx.reshape(4, 2))

        return image

    def _deskew(self, image):
        gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
        gray = cv2.bitwise_not(gray)
        thresh = cv2.threshold(
            gray, 0, 255, cv2.THRESH_BINARY | cv2.THRESH_OTSU
        )[1]

        coords = np.column_stack(np.where(thresh > 0))
        if len(coords) == 0:
            return image

        angle = cv2.minAreaRect(coords)[-1]
        if angle < -45:
            angle = -(90 + angle)
        else:
            angle = -angle

        (h, w) = image.shape[:2]
        center = (w // 2, h // 2)
        matrix = cv2.getRotationMatrix2D(center, angle, 1.0)
        rotated = cv2.warpAffine(
            image, matrix, (w, h),
            flags=cv2.INTER_CUBIC,
            borderMode=cv2.BORDER_REPLICATE
        )
        return rotated

    def _auto_rotate_orientation(self, image):
        try:
            pil_image = Image.fromarray(cv2.cvtColor(image, cv2.COLOR_BGR2RGB))
            osd = pytesseract.image_to_osd(pil_image)
            angle = 0

            for line in osd.split('\n'):
                if 'Rotate:' in line:
                    angle = int(line.split(':')[1].strip())
                    break

            if angle == 90:
                return cv2.rotate(image, cv2.ROTATE_90_COUNTERCLOCKWISE)
            if angle == 180:
                return cv2.rotate(image, cv2.ROTATE_180)
            if angle == 270:
                return cv2.rotate(image, cv2.ROTATE_90_CLOCKWISE)
        except Exception:
            pass

        return image

    def _preprocess_image(self, img):
        image = np.array(img)
        image = cv2.cvtColor(image, cv2.COLOR_RGB2BGR)

        # 1) Корекция на перспектива
        image = self._correct_perspective(image)

        # 2) Автоматично завъртане по ориентация
        image = self._auto_rotate_orientation(image)

        # 3) Изправяне (deskew)
        image = self._deskew(image)

        # Допълнителна OCR предобработка
        gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
        gray = cv2.resize(gray, None, fx=2, fy=2, interpolation=cv2.INTER_CUBIC)
        gray = cv2.GaussianBlur(gray, (3, 3), 0)
        gray = cv2.threshold(
            gray, 0, 255, cv2.THRESH_BINARY + cv2.THRESH_OTSU
        )[1]

        return Image.fromarray(gray)

    def post(self, request, *args, **kwargs):
        serializer = DocumentSerializer(data=request.data)

        if serializer.is_valid():
            document = serializer.save()

            try:
                img = Image.open(document.image.path).convert("RGB")
                img = self._preprocess_image(img)

                config = '--oem 1 --psm 4'
                extracted_text = pytesseract.image_to_string(
                    img,
                    lang='bul+eng',
                    config=config
                ).strip()

                document.extracted_text = extracted_text
                document.save()

                return Response({
                    'message': 'Документът е обработен успешно!',
                    'document_id': document.id,
                    'extracted_text': document.extracted_text,
                    'image_url': document.image.url
                }, status=status.HTTP_201_CREATED)

            except Exception as e:
                return Response(
                    {'error': f'Грешка при обработката на изображението: {str(e)}'},
                    status=status.HTTP_500_INTERNAL_SERVER_ERROR
                )

        return Response(serializer.errors, status=status.HTTP_400_BAD_REQUEST)
```

## Какво прави тази версия

### 1) Корекция на перспектива
Търси най-големия 4-ъгълен контур, предполага че това е документът, и го „изправя“.

### 2) Автоматично завъртане
Чете ориентацията на текста чрез `image_to_osd()` и завърта изображението при нужда.

### 3) Изправяне на наклона
Използва `deskew`, за да коригира леко завъртени снимки.

---

## Важно
Тази логика работи най-добре, когато:
- документът е ясно отделен от фона
- няма твърде много други обекти около него
- снимката не е много размазана

Ако фонът е сложен или документът не се разпознава като 4-ъгълник, корекцията на перспектива може да не се задейства — тогава кодът просто продължава с оригиналното изображение.

## Забележка
За този вариант ти трябват:
- `opencv-python`
- `numpy`

Ако искаш, мога да ти дам и **още по-добър вариант**, в който:
- се открива документът по по-надежден начин,
- се прави adaptive threshold,
- и има опция за **OCR само на български / само на английски / двата езика**.

Да, **може** — и това е много добра идея.  
След OCR обикновено е полезно да се направи **последващо почистване на текста**, защото Tesseract често връща:

- излишни нови редове
- прекъснати думи в края на реда
- двойни интервали
- „разпаднати“ абзаци
- грешки между подобни букви

## Какво можеш да направиш

### 1) Премахване на излишните нови редове
Най-често OCR връща текст като:

```text
Това е
примерен
текст
```

а ти искаш:

```text
Това е примерен текст
```

Но трябва да внимаваш, защото **не всички нови редове са грешка** — понякога са:
- край на абзац
- изброяване
- заглавие

Обикновено се прави правило:
- **единичен нов ред вътре в абзац** → заменя се с интервал
- **празен ред** → остава като разделител между абзаци

### 2) Премахване на пренасяне на думи
OCR често дава:

```text
дигитал-
ен скенер
```

Това може да се слее до:

```text
дигитален скенер
```

### 3) Нормализация на интервали и пунктуация
Например:
- да няма `текст  ,` вместо `текст,`
- да няма множество интервали
- да се оправят странни празни места около точка, запетая, двуеточие

## Последваща „смислова“ корекция
Да, **може да се прави и такава**, но с уговорка:

### Реалистично:
Можеш да направиш:
- корекция по **речник**
- корекция по **често срещани грешки**
- замяна на объркани символи
- проверка на думи срещу български/английски списък

### По-ограничено:
Истинска „смислова“ корекция, в смисъл автоматично да разбира контекста като човек, е вече по-сложна.  
Това обикновено изисква:
- езиков модел
- spell checker
- по-сложна NLP обработка

За дипломен проект е напълно достатъчно да кажеш, че правиш:

- **текстова нормализация**
- **речникова корекция**
- **премахване на структурни артефакти от OCR**

Това звучи професионално и е реалистично за реализация.

---

## Примерна стратегия
След OCR можеш да приложиш следните стъпки:

1. премахване на излишни интервали
2. сливане на редове в абзац
3. махане на пренасяния с тире
4. корекция на често грешни символи
5. евентуално проверка с речник

---

## Примерен Python код за постобработка

```python
import re

def postprocess_ocr_text(text):
    # 1) Премахване на пренасяне на думи с тире в края на реда
    text = re.sub(r'-\s*\n\s*', '', text)

    # 2) Замяна на единични нови редове с интервал,
    # но запазване на празните редове между абзаци
    text = re.sub(r'(?<!\n)\n(?!\n)', ' ', text)

    # 3) Нормализиране на многократни интервали
    text = re.sub(r'[ \t]+', ' ', text)

    # 4) Премахване на интервали преди пунктуация
    text = re.sub(r'\s+([,.;:!?])', r'\1', text)

    # 5) Нормализиране на празни редове
    text = re.sub(r'\n{3,}', '\n\n', text)

    return text.strip()
```

```python
from django.shortcuts import render
from rest_framework.views import APIView
from rest_framework.response import Response
from rest_framework import status
from .serializers import DocumentSerializer
import pytesseract
from PIL import Image
import cv2
import numpy as np
import re

pytesseract.pytesseract.tesseract_cmd = r'C:\Program Files\Tesseract-OCR\tesseract.exe'


class DocumentUploadView(APIView):
    def _order_points(self, pts):
        rect = np.zeros((4, 2), dtype="float32")
        s = pts.sum(axis=1)
        rect[0] = pts[np.argmin(s)]
        rect[2] = pts[np.argmax(s)]

        diff = np.diff(pts, axis=1)
        rect[1] = pts[np.argmin(diff)]
        rect[3] = pts[np.argmax(diff)]
        return rect

    def _four_point_transform(self, image, pts):
        rect = self._order_points(pts)
        (tl, tr, br, bl) = rect

        width_a = np.linalg.norm(br - bl)
        width_b = np.linalg.norm(tr - tl)
        max_width = max(int(width_a), int(width_b))

        height_a = np.linalg.norm(tr - br)
        height_b = np.linalg.norm(tl - bl)
        max_height = max(int(height_a), int(height_b))

        dst = np.array([
            [0, 0],
            [max_width - 1, 0],
            [max_width - 1, max_height - 1],
            [0, max_height - 1]
        ], dtype="float32")

        matrix = cv2.getPerspectiveTransform(rect, dst)
        warped = cv2.warpPerspective(image, matrix, (max_width, max_height))
        return warped

    def _correct_perspective(self, image):
        orig = image.copy()
        gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
        blurred = cv2.GaussianBlur(gray, (5, 5), 0)
        edged = cv2.Canny(blurred, 75, 200)

        contours, _ = cv2.findContours(edged.copy(), cv2.RETR_LIST, cv2.CHAIN_APPROX_SIMPLE)
        contours = sorted(contours, key=cv2.contourArea, reverse=True)[:5]

        for c in contours:
            peri = cv2.arcLength(c, True)
            approx = cv2.approxPolyDP(c, 0.02 * peri, True)

            if len(approx) == 4:
                return self._four_point_transform(orig, approx.reshape(4, 2))

        return image

    def _deskew(self, image):
        gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
        gray = cv2.bitwise_not(gray)
        thresh = cv2.threshold(
            gray, 0, 255, cv2.THRESH_BINARY | cv2.THRESH_OTSU
        )[1]

        coords = np.column_stack(np.where(thresh > 0))
        if len(coords) == 0:
            return image

        angle = cv2.minAreaRect(coords)[-1]
        if angle < -45:
            angle = -(90 + angle)
        else:
            angle = -angle

        (h, w) = image.shape[:2]
        center = (w // 2, h // 2)
        matrix = cv2.getRotationMatrix2D(center, angle, 1.0)
        rotated = cv2.warpAffine(
            image, matrix, (w, h),
            flags=cv2.INTER_CUBIC,
            borderMode=cv2.BORDER_REPLICATE
        )
        return rotated

    def _auto_rotate_orientation(self, image):
        try:
            pil_image = Image.fromarray(cv2.cvtColor(image, cv2.COLOR_BGR2RGB))
            osd = pytesseract.image_to_osd(pil_image)
            angle = 0

            for line in osd.split('\n'):
                if 'Rotate:' in line:
                    angle = int(line.split(':')[1].strip())
                    break

            if angle == 90:
                return cv2.rotate(image, cv2.ROTATE_90_COUNTERCLOCKWISE)
            if angle == 180:
                return cv2.rotate(image, cv2.ROTATE_180)
            if angle == 270:
                return cv2.rotate(image, cv2.ROTATE_90_CLOCKWISE)
        except Exception:
            pass

        return image

    def _postprocess_text(self, text):
        # Чести OCR корекции
        replacements = {
            "дигитал- ен": "дигитален",
            "документътт": "документът",
            "снимкаа": "снимка",
            "текстаа": "текста",
            "записваа": "записва",
            "приложениее": "приложение",
        }

        # 1) Махане на пренасяния на думи с тире
        text = re.sub(r'-\s*\n\s*', '', text)

        # 2) Сливане на единични нови редове в рамките на абзац
        text = re.sub(r'(?<!\n)\n(?!\n)', ' ', text)

        # 3) Нормализиране на интервали
        text = re.sub(r'[ \t]+', ' ', text)

        # 4) Премахване на интервали преди пунктуация
        text = re.sub(r'\s+([,.;:!?])', r'\1', text)

        # 5) Нормализиране на повече от 2 празни реда
        text = re.sub(r'\n{3,}', '\n\n', text)

        # 6) Прилагане на чести ръчни замени
        for wrong, correct in replacements.items():
            text = text.replace(wrong, correct)

        return text.strip()

    def _preprocess_image(self, img):
        image = np.array(img)
        image = cv2.cvtColor(image, cv2.COLOR_RGB2BGR)

        # 1) Корекция на перспектива
        image = self._correct_perspective(image)

        # 2) Автоматично завъртане по ориентация
        image = self._auto_rotate_orientation(image)

        # 3) Изправяне (deskew)
        image = self._deskew(image)

        # Допълнителна OCR предобработка
        gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
        gray = cv2.resize(gray, None, fx=2, fy=2, interpolation=cv2.INTER_CUBIC)
        gray = cv2.GaussianBlur(gray, (3, 3), 0)
        gray = cv2.threshold(
            gray, 0, 255, cv2.THRESH_BINARY + cv2.THRESH_OTSU
        )[1]

        return Image.fromarray(gray)

    def post(self, request, *args, **kwargs):
        serializer = DocumentSerializer(data=request.data)

        if serializer.is_valid():
            document = serializer.save()

            try:
                img = Image.open(document.image.path).convert("RGB")
                img = self._preprocess_image(img)

                config = '--oem 1 --psm 4'
                extracted_text = pytesseract.image_to_string(
                    img,
                    lang='bul+eng',
                    config=config
                ).strip()

                extracted_text = self._postprocess_text(extracted_text)

                document.extracted_text = extracted_text
                document.save()

                return Response({
                    'message': 'Документът е обработен успешно!',
                    'document_id': document.id,
                    'extracted_text': document.extracted_text,
                    'image_url': document.image.url
                }, status=status.HTTP_201_CREATED)

            except Exception as e:
                return Response(
                    {'error': f'Грешка при обработката на изображението: {str(e)}'},
                    status=status.HTTP_500_INTERNAL_SERVER_ERROR
                )

        return Response(serializer.errors, status=status.HTTP_400_BAD_REQUEST)
```

## Какво е по-интелигентното тук
- **Не слепва безразборно всички редове**
- Запазва **празни редове като разделител на абзаци**
- Не смесва лесно:
  - заглавия
  - списъци
  - отделни линии
- Слива редове, когато е вероятно да са част от един абзац
- Премахва преноси с тире като `информа-` + `ция`

## Ако искаш още по-добър резултат
Мога да направя и следваща версия с:
- **разпознаване на булети и номерирани списъци**
- **запазване на заглавията на отделен ред**
- **по-умна корекция по речник на български думи**
- или **отделна Django функция/utility файл**, за да не се трупа логиката във `views.py`

Ако искаш, в следващото съобщение мога да ти дам и **вариант, разделен в отделен helper файл**, за да стане кодът по-чист.

```javascript
createApp({
    setup() {
        const selectedFile = ref(null);
        const previewUrl = ref(null);
        const extractedText = ref('');
        const isLoading = ref(false);
        const errorMessage = ref('');

        // При избиране на файл
        const onFileChange = (event) => {
            const file = event.target.files[0];
            if (file) {
                selectedFile.value = file;
                previewUrl.value = URL.createObjectURL(file);
                extractedText.value = '';
                errorMessage.value = '';
            }
        };

        // Изпращане към бекенда
        const uploadDocument = async () => {
            if (!selectedFile.value) return;

            isLoading.value = true;
            errorMessage.value = '';

            const formData = new FormData();
            formData.append('image', selectedFile.value);

            try {
                const response = await axios.post('/api/upload/', formData, {
                    headers: {
                        'Content-Type': 'multipart/form-data'
                    }
                });

                extractedText.value = response.data.extracted_text || '';

            } catch (error) {
                console.error(error);
                errorMessage.value = "Възникна грешка при обработката на изображението.";
            } finally {
                isLoading.value = false;
            }
        };

        const downloadTxt = () => {
            const text = extractedText.value || '';
            const blob = new Blob([text], { type: 'text/plain;charset=utf-8' });
            const url = URL.createObjectURL(blob);

            const link = document.createElement('a');
            link.href = url;
            link.download = 'ocr_text.txt';
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);

            URL.revokeObjectURL(url);
        };

        return {
            selectedFile,
            previewUrl,
            extractedText,
            isLoading,
            errorMessage,
            onFileChange,
            uploadDocument,
            downloadTxt
        }
    }
}).mount('#app');
```

## Какво правят тези промени

### 1) Камера от устройството
Добавихме:

```html
capture="environment"
```

Това казва на браузъра да отвори **камерата на телефона** или подходящ източник за заснемане.  
На мобилни устройства обикновено ще се появи избор за камера.

### 2) Изтегляне като `.txt`
След като потребителят редактира текста в `textarea`, бутонът **„Изтегли като .txt“** ще изтегли текущото съдържание като текстов файл.

---

## Важно уточнение
`capture="environment"` работи най-добре на **телефони**.  
На лаптоп/десктоп:
- може да отвори избор на файл,
- или камерата, ако браузърът и устройството го поддържат.

Ако искаш, в следващата стъпка мога да добавя и **истинска уеб камера с live preview и бутон „Снимай“** чрез `getUserMedia`, което е по-професионално и работи по-добре на лаптопи.

```javascript
const { createApp, ref } = Vue;

createApp({
    setup() {
        const selectedFile = ref(null);
        const previewUrl = ref(null);
        const extractedText = ref('');
        const isLoading = ref(false);
        const errorMessage = ref('');

        // При избиране на файл
        const onFileChange = (event) => {
            const file = event.target.files[0];
            if (file) {
                selectedFile.value = file;
                previewUrl.value = URL.createObjectURL(file);
                extractedText.value = ''; // Изчистваме стария текст
                errorMessage.value = '';
            }
        };

        // Изпращане към бекенда
        const uploadDocument = async () => {
            if (!selectedFile.value) return;

            isLoading.value = true;
            errorMessage.value = '';

            // Използваме FormData, защото изпращаме файл
            const formData = new FormData();
            formData.append('image', selectedFile.value);

            try {
                // Изпращаме POST заявка към създадения API endpoint
                const response = await axios.post('/api/upload/', formData, {
                    headers: {
                        'Content-Type': 'multipart/form-data'
                    }
                });

                extractedText.value = response.data.extracted_text || '';

            } catch (error) {
                console.error(error);
                errorMessage.value = "Възникна грешка при обработката на изображението.";
            } finally {
                isLoading.value = false;
            }
        };

        // Изтегляне на текста като .txt файл
        const downloadTxt = () => {
            const text = extractedText.value || '';
            const blob = new Blob([text], { type: 'text/plain;charset=utf-8' });
            const url = URL.createObjectURL(blob);

            const link = document.createElement('a');
            link.href = url;
            link.download = 'ocr_text.txt';
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);

            URL.revokeObjectURL(url);
        };

        return {
            selectedFile,
            previewUrl,
            extractedText,
            isLoading,
            errorMessage,
            onFileChange,
            uploadDocument,
            downloadTxt
        }
    }
}).mount('#app');
```

```plain text
1.	Тема:
Оптическо разпознаване на символи и дигитализация. Разработване на приложение „Дигитален скенер“, при което потребителят качва снимка на документ, а системата извлича текста чрез OCR и го записва в база данни и/или го предлага за редакция в HTML форма.
2.	Входни данни:
Дигитализацията на документи чрез снимка от телефон или сканирано изображение е широко използван подход за автоматизирано въвеждане и архивиране на информация (фактури, бележки, договори, формуляри). Основната технология за това е OCR (Optical Character Recognition), която преобразува текст от изображение в машинно-четим вид. Качеството на разпознаването зависи от фактори като осветеност, наклон/перспектива, резолюция, шум, фон и тип шрифт, поради което често се прилагат техники за предобработка на изображението (корекция на контраст, изправяне, бинаризация и др.). Разработването на приложение, което приема изображение на документ, извлича текст, позволява корекция от потребителя и съхранява резултата в база данни, представлява актуална и практически приложима задача.
```

```javascript
const { createApp, ref, computed, onBeforeUnmount } = Vue;

createApp({
    setup() {
        const selectedFile = ref(null);
        const previewUrl = ref(null);
        const extractedText = ref('');
        const isLoading = ref(false);
        const errorMessage = ref('');
        const successMessage = ref('');

        const cameraStream = ref(null);
        const cameraActive = ref(false);
        const cameraPreview = ref(null);
        const capturedFromCamera = ref(false);

        const canUseCamera = computed(() => !!(navigator.mediaDevices && navigator.mediaDevices.getUserMedia));

        const stopCamera = () => {
            if (cameraStream.value) {
                cameraStream.value.getTracks().forEach(track => track.stop());
                cameraStream.value = null;
            }
            cameraActive.value = false;
        };

        const onFileChange = (event) => {
            const file = event.target.files[0];
            if (file) {
                selectedFile.value = file;
                previewUrl.value = URL.createObjectURL(file);
                extractedText.value = '';
                errorMessage.value = '';
                successMessage.value = '';
                capturedFromCamera.value = false;
            }
        };

        const startCamera = async () => {
            errorMessage.value = '';
            successMessage.value = '';

            try {
                stopCamera();

                const stream = await navigator.mediaDevices.getUserMedia({
                    video: {
                        facingMode: 'environment'
                    },
                    audio: false
                });

                cameraStream.value = stream;
                cameraActive.value = true;

                if (cameraPreview.value) {
                    cameraPreview.value.srcObject = stream;
                    await cameraPreview.value.play();
                }
            } catch (error) {
                console.error(error);
                errorMessage.value = 'Не може да се стартира камерата. Провери разрешенията на браузъра.';
            }
        };

        const capturePhoto = () => {
            errorMessage.value = '';
            successMessage.value = '';

            if (!cameraPreview.value) {
                errorMessage.value = 'Камерата не е активна.';
                return;
            }

            const video = cameraPreview.value;
            const canvas = document.createElement('canvas');
            canvas.width = video.videoWidth;
            canvas.height = video.videoHeight;

            if (!canvas.width || !canvas.height) {
                errorMessage.value = 'Камерата още не е готова.';
                return;
            }

            const ctx = canvas.getContext('2d');
            ctx.drawImage(video, 0, 0, canvas.width, canvas.height);

            canvas.toBlob((blob) => {
                if (!blob) {
                    errorMessage.value = 'Неуспешно заснемане на изображението.';
                    return;
                }

                const file = new File([blob], `camera_${Date.now()}.jpg`, {
                    type: 'image/jpeg'
                });

                selectedFile.value = file;
                previewUrl.value = URL.createObjectURL(blob);
                extractedText.value = '';
                capturedFromCamera.value = true;
                successMessage.value = 'Снимката е заснета успешно.';
            }, 'image/jpeg', 0.95);
        };

        const uploadDocument = async () => {
            if (!selectedFile.value) {
                errorMessage.value = 'Моля, избери или заснеми изображение първо.';
                return;
            }

            isLoading.value = true;
            errorMessage.value = '';
            successMessage.value = '';

            const formData = new FormData();
            formData.append('image', selectedFile.value);

            try {
                const response = await axios.post('/api/upload/', formData, {
                    headers: {
                        'Content-Type': 'multipart/form-data'
                    }
                });

                extractedText.value = response.data.extracted_text || '';
                successMessage.value = 'OCR обработката е завършена успешно.';
            } catch (error) {
                console.error(error);
                errorMessage.value = 'Възникна грешка при обработката на изображението.';
            } finally {
                isLoading.value = false;
            }
        };

        const downloadTxt = () => {
            const text = extractedText.value || '';
            if (!text.trim()) {
                errorMessage.value = 'Няма текст за изтегляне.';
                return;
            }

            const blob = new Blob([text], { type: 'text/plain;charset=utf-8' });
            const url = URL.createObjectURL(blob);

            const link = document.createElement('a');
            link.href = url;
            link.download = 'ocr_text.txt';
            link.style.display = 'none';

            document.body.appendChild(link);
            link.click();

            setTimeout(() => {
                document.body.removeChild(link);
                URL.revokeObjectURL(url);
            }, 100);
        };

        onBeforeUnmount(() => {
            stopCamera();
        });

        return {
            selectedFile,
            previewUrl,
            extractedText,
            isLoading,
            errorMessage,
            successMessage,
            cameraStream,
            cameraActive,
            cameraPreview,
            canUseCamera,
            onFileChange,
            startCamera,
            capturePhoto,
            uploadDocument,
            downloadTxt,
            stopCamera
        };
    }
}).mount('#app');
```

## Какво добавихме

### 1) Истинска уеб камера
На компютър/лаптоп можеш да:
- стартираш камерата,
- видиш live preview,
- да натиснеш **„Снимай“**,
- после да пратиш снимката към OCR.

### 2) Телефонът не се разваля
Оставихме и старото:
- избор на файл,
- `capture="environment"` за телефон.

Тоест на мобилен телефон потребителят може да ползва и вградения camera picker, а ако браузърът поддържа — и директната камера.

### 3) По-надеждно изтегляне на `.txt`
Добавих:
- проверка дали има текст,
- по-стабилно създаване и премахване на временния `Blob` URL.

---

## Важно за камерата
За да работи `getUserMedia`:
- страницата трябва да е през **HTTPS**, или
- да е на **localhost**.

Ако я пускаш локално в разработка, това обикновено е ОК.

---

Мога да добавя и:
- **бутон „Заснеми и OCR“** в една стъпка;
- **автоматично OCR след заснемане**;
- **избор между файл / камера** с по-ясен UI;
- или **запазване на `.txt` и от бекенда**, ако искаш да се сваля като генериран файл от Django.

```javascript
        const downloadTxt = () => {
            const text = extractedText.value || '';
            if (!text.trim()) {
                errorMessage.value = 'Няма текст за изтегляне.';
                return;
            }

            const blob = new Blob([text], { type: 'text/plain;charset=utf-8' });
            const url = URL.createObjectURL(blob);

            const link = document.createElement('a');
            link.href = url;
            link.download = 'ocr_text.txt';
            link.style.display = 'none';

            document.body.appendChild(link);
            link.click();

            setTimeout(() => {
                document.body.removeChild(link);
                URL.revokeObjectURL(url);
            }, 100);
        };

        onMounted(() => {
            detectCameraAvailability();
        });

        onBeforeUnmount(() => {
            stopCamera();
        });

        return {
            selectedFile,
            previewUrl,
            extractedText,
            isLoading,
            errorMessage,
            successMessage,
            cameraStream,
            cameraActive,
            cameraPreview,
            canUseCamera,
            onFileChange,
            startCamera,
            capturePhoto,
            uploadDocument,
            downloadTxt,
            stopCamera
        };
    }
}).mount('#app');
```

## Какво постигаме с това

- На **телефон** всичко си остава както е.
- На **компютър без камера** бутонът няма да е видим.
- Ако въпреки това някой извика камерата, ще се покаже ясно съобщение:
  - „Не е открита камера на това устройство.“
- Няма да се пълни конзолата с неясни runtime грешки.

## За бутона „Изтегли“
Това, че не се случва нищо на компютър, често е защото:
- няма разпознат текст;
- или браузърът блокира изтегляне без реално съдържание;
- или бутонът е натиснат преди OCR да е приключил.

В текущата версия вече има проверка `if (!text.trim())`, което е добре.  
Ако искаш, мога да направя и **по-видим бутон/съобщение за изтегляне**, плюс **показване на наличната камера като отделен статус**, за да е интерфейсът по-ясен.
